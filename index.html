const express = require('express'); const session = require('express-session'); const path = require('path'); const app = express(); const PORT = process.env.PORT || 3000; // Middleware app.use(express.urlencoded({ extended: true })); app.use(express.json()); // for delete request app.use(express.static(path.join(__dirname, 'public'))); app.use(session({ secret: 'projectx-ultra-glass-secret', resave: false, saveUninitialized: true, cookie: { maxAge: 30 * 24 * 60 * 60 * 1000 } })); // In-memory storage let users = []; // { username } let questions = []; // { id, author, text, timestamp, replies: [{ author, text, timestamp }] } // Helper function isUsernameUnique(username) { return !users.some(u => u.username === username); } function formatTime(isoString) { return new Date(isoString).toLocaleString(); } function escapeHtml(str) { return str.replace(/[&<>]/g, m => ({ '&': '&', '<': '<', '>': '>' })[m]); } // Serve HTML app.get('/', (req, res) => { res.sendFile(path.join(__dirname, 'index.html')); }); // API endpoints app.post('/register', (req, res) => { const username = req.body.username?.trim(); if (!username) return res.redirect('/?error=empty'); if (!isUsernameUnique(username)) return res.redirect('/?error=taken'); users.push({ username }); req.session.username = username; res.redirect('/'); }); app.post('/change-nick', (req, res) => { if (!req.session.username) return res.redirect('/'); const newUsername = req.body.newUsername?.trim(); if (!newUsername) return res.redirect('/'); if (newUsername !== req.session.username && !isUsernameUnique(newUsername)) { return res.redirect('/?error=nickTaken'); } const oldName = req.session.username; // Update in users array const userIdx = users.findIndex(u => u.username === oldName); if (userIdx !== -1) users[userIdx].username = newUsername; // Update in questions and replies questions.forEach(q => { if (q.author === oldName) q.author = newUsername; q.replies.forEach(r => { if (r.author === oldName) r.author = newUsername; }); }); req.session.username = newUsername; res.redirect('/'); }); app.post('/ask', (req, res) => { if (!req.session.username) return res.redirect('/'); const text = req.body.question?.trim(); if (!text) return res.redirect('/'); questions.push({ id: Date.now().toString(), author: req.session.username, text, timestamp: new Date().toISOString(), replies: [] }); res.redirect('/'); }); app.post('/reply/:id', (req, res) => { if (!req.session.username) return res.redirect('/'); const questionId = req.params.id; const replyText = req.body.reply?.trim(); if (!replyText) return res.redirect('/'); const q = questions.find(q => q.id === questionId); if (q) { q.replies.push({ author: req.session.username, text: replyText, timestamp: new Date().toISOString() }); } res.redirect('/'); }); app.post('/delete-question', (req, res) => { if (!req.session.username) return res.status(403).json({ error: 'Unauthorized' }); const { id } = req.body; const qIndex = questions.findIndex(q => q.id === id); if (qIndex === -1) return res.status(404).json({ error: 'Not found' }); // Only author can delete if (questions[qIndex].author !== req.session.username) { return res.status(403).json({ error: 'Forbidden' }); } questions.splice(qIndex, 1); res.json({ success: true }); }); // Expose data for client (optional, but helps) app.get('/api/data', (req, res) => { if (!req.session.username) return res.status(401).json({ error: 'Not logged in' }); res.json({ currentUser: req.session.username, users: users.map(u => u.username), questions: questions.slice().reverse() // newest first }); }); app.listen(PORT, () => { console.log(`🚀 Project X running at http://localhost:${PORT}`); });
