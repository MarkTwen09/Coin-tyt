const express = require('express'); const session = require('express-session'); const path = require('path'); const app = express(); const PORT = process.env.PORT || 3000; // Middleware app.use(express.urlencoded({ extended: true })); app.use(express.static(path.join(__dirname, 'public'))); app.use(session({ secret: 'projectx-secret-key-2024', resave: false, saveUninitialized: true, cookie: { maxAge: 30 * 24 * 60 * 60 * 1000 } // 30 days })); // In‑memory storage let messages = []; // { username, text, timestamp } let registeredUsers = []; // { username, registeredAt } // Helper to check uniqueness function isUsernameUnique(username) { return !registeredUsers.some(u => u.username === username); } // Helper to format timestamp function formatTime(isoString) { const date = new Date(isoString); return date.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' }); } // Home page – main chat app.get('/', (req, res) => { if (!req.session.username) { // Not logged in → show registration form return res.send(` Project X – Регистрация

Project X

${req.query.error}
` : ''}
`); } // User is logged in – render main chat page const currentUser = req.session.username; const onlineUsers = registeredUsers.map(u => u.username).filter(u => u !== currentUser); // Limit to last 100 messages const recentMessages = messages.slice(-100).reverse(); // newest last? We'll show chronological in template res.send(` Project X – Liquid Glass Messenger
${currentUser.charAt(0).toUpperCase()}

`).join('')} ${messages.length === 0 ? '
Нет сообщений. Напишите первое!
' : ''}
`); }); // Registration app.post('/register', (req, res) => { const username = req.body.username.trim(); if (!username) { return res.redirect('/?error=Имя+не+может+быть+пустым'); } if (!isUsernameUnique(username)) { return res.redirect('/?error=Этот+никнейм+уже+занят'); } registeredUsers.push({ username, registeredAt: new Date().toISOString() }); req.session.username = username; res.redirect('/'); }); // Send message app.post('/send', (req, res) => { if (!req.session.username) return res.redirect('/'); const text = req.body.message?.trim(); if (!text) return res.redirect('/'); messages.push({ username: req.session.username, text: text, timestamp: new Date().toISOString() }); // Keep last 100 messages if (messages.length > 100) messages.shift(); res.redirect('/'); }); // Update profile (change nickname) app.post('/update-profile', (req, res) => { if (!req.session.username) return res.redirect('/'); const newUsername = req.body.newUsername?.trim(); if (!newUsername) return res.redirect('/'); // Cannot change to an already existing username (except own) if (newUsername !== req.session.username && !isUsernameUnique(newUsername)) { return res.redirect('/?error=Никнейм+уже+занят'); } // Update in registeredUsers const userIndex = registeredUsers.findIndex(u => u.username === req.session.username); if (userIndex !== -1) registeredUsers[userIndex].username = newUsername; // Update session req.session.username = newUsername; res.redirect('/'); }); // Helper function for escaping function escapeHtml(str) { return str.replace(/[&<>]/g, (m) => { if (m === '&') return '&'; if (m === '<') return '<'; if (m === '>') return '>'; return m; }); } // Format time helper used in template function formatTime(isoString) { const date = new Date(isoString); return date.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' }); } // Expose helper to template via res.locals? We'll pass it manually in render. // In the res.send we used a function; we need to make it available. // Since we are using string concatenation, we have to pass the function as a variable in the string. // In the above I used formatTime inside the map, but it's not defined in the browser – it's used on server side. // The template is built on server, so formatTime is a server function and works. app.listen(PORT, () => { console.log(`🚀 Project X server running at http://localhost:${PORT}`); });
