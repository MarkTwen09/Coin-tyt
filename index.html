const express = require('express'); const session = require('express-session'); const path = require('path'); const app = express(); const PORT = process.env.PORT || 3000; // Middleware app.use(express.urlencoded({ extended: true })); app.use(express.static(path.join(__dirname, 'public'))); app.use(session({ secret: 'projectx-qa-secret-key', resave: false, saveUninitialized: true, cookie: { maxAge: 30 * 24 * 60 * 60 * 1000 } // 30 days })); // In‑memory storage let users = []; // { username } let questions = []; // { id, author, text, timestamp, replies: [{ author, text, timestamp }] } // Helper to check if username exists function isUsernameUnique(username) { return !users.some(u => u.username === username); } // Helper to escape HTML function escapeHtml(str) { return str.replace(/[&<>]/g, (m) => { if (m === '&') return '&'; if (m === '<') return '<'; if (m === '>') return '>'; return m; }); } // Helper to format timestamp function formatTime(isoString) { const date = new Date(isoString); return date.toLocaleString(); } // Home page app.get('/', (req, res) => { if (!req.session.username) { // Not logged in → show registration form return res.send(` Project X – Вход

Project X

${req.query.error ? `
${escapeHtml(req.query.error)}
` : ''}
`); } // User is logged in – render main Q&A page const currentUser = req.session.username; // Sort questions: newest first const sortedQuestions = [...questions].reverse(); res.send(` Project X – Вопросы и ответы
${escapeHtml(currentUser.charAt(0).toUpperCase())}

${escapeHtml(currentUser)}

Ваш публичный профиль

${sortedQuestions.map(q => `
${escapeHtml(q.author.charAt(0).toUpperCase())}
${escapeHtml(q.author)}
${formatTime(q.timestamp)}
${escapeHtml(q.text)}
${q.replies && q.replies.length ? q.replies.map(r => `
${escapeHtml(r.author.charAt(0).toUpperCase())}
${escapeHtml(r.author)}
${escapeHtml(r.text)}
${formatTime(r.timestamp)}
`).join('') : '
Нет ответов. Будьте первым!
'}
`).join('')} ${questions.length === 0 ? '
Пока нет вопросов. Задайте первый!
' : ''}
`); }); // Register new user app.post('/register', (req, res) => { const username = req.body.username.trim(); if (!username) { return res.redirect('/?error=Имя+не+может+быть+пустым'); } if (!isUsernameUnique(username)) { return res.redirect('/?error=Этот+никнейм+уже+занят'); } users.push({ username }); req.session.username = username; res.redirect('/'); }); // Change nickname app.post('/change-nick', (req, res) => { if (!req.session.username) return res.redirect('/'); const newUsername = req.body.newUsername.trim(); if (!newUsername) return res.redirect('/'); // Check uniqueness (excluding current user) const currentUser = req.session.username; if (newUsername !== currentUser && !isUsernameUnique(newUsername)) { return res.redirect('/?error=Никнейм+уже+занят'); } // Update in users array const userIndex = users.findIndex(u => u.username === currentUser); if (userIndex !== -1) users[userIndex].username = newUsername; // Update in questions and replies questions.forEach(q => { if (q.author === currentUser) q.author = newUsername; q.replies.forEach(r => { if (r.author === currentUser) r.author = newUsername; }); }); req.session.username = newUsername; res.redirect('/'); }); // Ask a question app.post('/ask', (req, res) => { if (!req.session.username) return res.redirect('/'); const text = req.body.question.trim(); if (!text) return res.redirect('/'); const newQuestion = { id: Date.now().toString(), author: req.session.username, text: text, timestamp: new Date().toISOString(), replies: [] }; questions.push(newQuestion); // Keep last 100 questions if (questions.length > 100) questions.shift(); res.redirect('/'); }); // Reply to a question app.post('/reply/:questionId', (req, res) => { if (!req.session.username) return res.redirect('/'); const questionId = req.params.questionId; const replyText = req.body.reply.trim(); if (!replyText) return res.redirect('/'); const question = questions.find(q => q.id === questionId); if (question) { question.replies.push({ author: req.session.username, text: replyText, timestamp: new Date().toISOString() }); // Keep only last 20 replies per question to avoid bloating if (question.replies.length > 20) question.replies.shift(); } res.redirect('/'); }); // Start server app.listen(PORT, () => { console.log(`🚀 Project X Q&A server running at http://localhost:${PORT}`); });
