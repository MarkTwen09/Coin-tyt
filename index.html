const express = require('express'); const http = require('http'); const { Server } = require('socket.io'); const path = require('path'); const app = express(); const server = http.createServer(app); const io = new Server(server); // Хранилище данных let users = []; // { socketId, username, isOnline } let messages = []; // { username, text, timestamp } // Раздача статики app.use(express.static(path.join(__dirname))); // Сокет-соединения io.on('connection', (socket) => { console.log(`🔌 Новое подключение: ${socket.id}`); // Регистрация пользователя socket.on('register', (username) => { // Проверка уникальности const existing = users.find(u => u.username === username); if (existing) { socket.emit('register_error', 'Этот никнейм уже занят'); return; } // Добавляем пользователя const newUser = { socketId: socket.id, username: username, isOnline: true }; users.push(newUser); socket.username = username; // Отправляем историю сообщений и список пользователей новичку socket.emit('history', messages); socket.emit('users_list', users.filter(u => u.isOnline).map(u => u.username)); // Оповещаем всех об обновлении списка онлайн io.emit('users_list', users.filter(u => u.isOnline).map(u => u.username)); socket.broadcast.emit('user_joined', username); console.log(`✅ Пользователь ${username} зарегистрирован`); }); // Получение нового сообщения socket.on('new_message', (data) => { const { text } = data; const username = socket.username; if (!username) return; const message = { username, text, timestamp: new Date().toISOString(), }; messages.push(message); // Ограничим историю последними 100 сообщениями if (messages.length > 100) messages.shift(); // Рассылаем сообщение всем io.emit('new_message', message); }); // Индикатор "печатает" socket.on('typing_start', () => { if (socket.username) { socket.broadcast.emit('user_typing', socket.username); } }); socket.on('typing_stop', () => { if (socket.username) { socket.broadcast.emit('user_stop_typing', socket.username); } }); // Отключение socket.on('disconnect', () => { if (socket.username) { // Удаляем из списка users = users.filter(u => u.socketId !== socket.id); io.emit('users_list', users.filter(u => u.isOnline).map(u => u.username)); io.emit('user_left', socket.username); console.log(`❌ Пользователь ${socket.username} отключился`); } }); }); const PORT = process.env.PORT || 3000; server.listen(PORT, () => { console.log(`🚀 Сервер запущен на http://localhost:${PORT}`); }); Project X | Telegram‑like мессенджер
