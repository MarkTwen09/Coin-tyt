const express = require('express'); const http = require('http'); const { Server } = require('socket.io'); const path = require('path'); const app = express(); const server = http.createServer(app); const io = new Server(server); // Хранилище данных let users = []; // { socketId, username, displayName, isOnline } let messages = []; // { username, displayName, text, timestamp } // Раздача статики app.use(express.static(path.join(__dirname))); // Сокет-соединения io.on('connection', (socket) => { console.log(`🔌 Новое подключение: ${socket.id}`); // Регистрация пользователя socket.on('register', (data) => { const { username, displayName } = data; // Проверка уникальности username const existing = users.find(u => u.username === username); if (existing) { socket.emit('register_error', 'Этот юзернейм уже занят'); return; } // Добавляем пользователя const newUser = { socketId: socket.id, username, displayName: displayName || username, isOnline: true }; users.push(newUser); socket.username = username; // Отправляем историю сообщений и список пользователей новичку socket.emit('history', messages); socket.emit('users_list', users.filter(u => u.isOnline).map(u => ({ username: u.username, displayName: u.displayName }))); // Оповещаем всех об обновлении списка онлайн io.emit('users_list', users.filter(u => u.isOnline).map(u => ({ username: u.username, displayName: u.displayName }))); socket.broadcast.emit('user_joined', { username: newUser.username, displayName: newUser.displayName }); console.log(`✅ Пользователь ${newUser.username} (${newUser.displayName}) зарегистрирован`); }); // Обновление профиля socket.on('update_user', (data, callback) => { const { newUsername, newDisplayName } = data; const user = users.find(u => u.socketId === socket.id); if (!user) { callback({ success: false, error: 'Пользователь не найден' }); return; } // Проверяем уникальность нового юзернейма, если он меняется if (newUsername && newUsername !== user.username) { const taken = users.some(u => u.username === newUsername && u.socketId !== socket.id); if (taken) { callback({ success: false, error: 'Юзернейм уже занят' }); return; } user.username = newUsername; } if (newDisplayName) { user.displayName = newDisplayName; } // Сохраняем обновлённые данные callback({ success: true, user: { username: user.username, displayName: user.displayName } }); // Рассылаем обновлённый список пользователей io.emit('users_list', users.filter(u => u.isOnline).map(u => ({ username: u.username, displayName: u.displayName }))); // Оповещаем всех, что пользователь обновил профиль (опционально) socket.broadcast.emit('user_updated', { username: user.username, displayName: user.displayName }); console.log(`✏️ Пользователь ${user.username} обновил профиль`); }); // Получение нового сообщения socket.on('new_message', (data) => { const { text } = data; const user = users.find(u => u.socketId === socket.id); if (!user) return; const message = { username: user.username, displayName: user.displayName, text, timestamp: new Date().toISOString(), }; messages.push(message); // Ограничим историю последними 100 сообщениями if (messages.length > 100) messages.shift(); // Рассылаем сообщение всем io.emit('new_message', message); }); // Индикатор "печатает" socket.on('typing_start', () => { const user = users.find(u => u.socketId === socket.id); if (user) { socket.broadcast.emit('user_typing', user.displayName); } }); socket.on('typing_stop', () => { const user = users.find(u => u.socketId === socket.id); if (user) { socket.broadcast.emit('user_stop_typing', user.displayName); } }); // Отключение socket.on('disconnect', () => { const user = users.find(u => u.socketId === socket.id); if (user) { users = users.filter(u => u.socketId !== socket.id); io.emit('users_list', users.filter(u => u.isOnline).map(u => ({ username: u.username, displayName: u.displayName }))); io.emit('user_left', { username: user.username, displayName: user.displayName }); console.log(`❌ Пользователь ${user.username} (${user.displayName}) отключился`); } }); }); const PORT = process.env.PORT || 3000; server.listen(PORT, () => { console.log(`🚀 Сервер запущен на http://localhost:${PORT}`); }); Project X | Telegram‑like мессенджер
